165 double gx(double x, double gamma, double mu_site, double L){
166  if(gamma != 0) return 2*mu_site*L*(1-exp(-1*gamma*(1-x)))/((1-exp(-1*gamma))*x*(1-x));
167  return 2*mu_site*L/x;
168 }

171 double* G(double gamma,double mu_site, double L, double N_chrome){
172  double total_SNPs = 0;
173  double* g = new double[(int)N_chrome];
174 
175  for(int j = 1; j <= (N_chrome - 1); j++){
176  double freq = j/(N_chrome);
177  g[j] = gx(freq, gamma, mu_site, L);
178  total_SNPs += g[j];
179  }
180 
181  g[0] = L-total_SNPs;
182 
183  return g;
